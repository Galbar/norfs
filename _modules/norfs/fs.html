    def __eq__(self, other: Any) -> bool:
        if other is self:
            return True
        if not isinstance(other, DirListResult):
            return False
        if set(other.files) != set(self.files):
            return False
        if set(other.dirs) != set(self.dirs):
            return False
        if set(other.others) != set(self.others):
            return False
        return True
